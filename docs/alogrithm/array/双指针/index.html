    左右指针

    • 左右指针分别指向数组第一个元素和数组最后一个元素
      • num[left]+num[right]>target:right--;
      • num[left]+num[right]<target:left++;
    public int[] twoSum(int[] numbers, int target) {
    int left = 0;
    int right = numbers.length - 1;
    while (left < right) {
      int p = numbers[left] + numbers[right];
      if (p == target) {
        return new int[]{left+1, right+1};
      } else if (p > target) {
        right--;
      } else {
        left++;
      }
    }
    return null;
}

    two sum 改造

    • 增加一个搜索范围, lo, hi
    • 成功找到一组后, lo 从下一个不同的地方开始
    public List<List<Integer>> twoSum(int[] nums, int lo, int hi, int target) {
    List<List<Integer>> result = new ArrayList<>();
    int left = lo;
    int right = hi;
    while (left < right) {
        int nleft = nums[left];
        int nright = nums[right];
        int p = nleft + nright;
        if (p == target) {
            List<Integer> tmp = new ArrayList<>(Arrays.asList(nleft, nright, nums[lo-1]));
            result.add(tmp);
            while (nums[left] == nleft && left < right) {
              left++;
            }
        } else if (p > target) {
            right--;
        } else {
            left++;
        }
    }
    return result;
}

public List<List<Integer>> threeSum(int[] nums) {
    List<List<Integer>> result = new ArrayList<>();
    Arrays.sort(nums);
    for (int i = 0; i < nums.length-2; i++) {
        int num = nums[i];
        List<List<Integer>> twoSumResult = twoSum(nums, i+1, nums.length-1, 0-num);
        if (!twoSumResult.isEmpty()) {
            result.addAll(twoSumResult);
            while (i+1 < nums.length-2 && num == nums[i+1]) {
              i++;
            }
        }
    }
    return result;
}
